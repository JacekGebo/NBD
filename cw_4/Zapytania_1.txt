Zad. 2
MATCH (t:town{name: "Darjeeling"}),
      (pk:peak{name: "Sandakphu"}),
      p = shortestPath((t)-[*]-(pk))
WHERE all(r IN relationships(p) WHERE r.winter = "true")
RETURN p

Zad. 3
MATCH (t:town{name: "Darjeeling"}),
      (pk:peak{name: "Sandakphu"}),
      p = shortestPath((t)-[*]-(pk))
WITH p ORDER BY all(r IN relationships(p) ORDER BY r.distance)
RETURN p
